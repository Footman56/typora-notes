```
# 查询好确认的时间
SELECT  `todo`.`plan_id` , plan.`plan_name` , case  
  WHen plan.`plan_period` = 6 then '第三季度' 
  when plan.`plan_period` = 7 then '第四季度'
  when plan.`plan_period` = 30 then '十一月'
  when plan.`plan_period` = 31 then '十二月'
   end as '考核周期' ,
 emm.employee_name   as  '被考核人', emp.`employee_name` as '待办人' , from_unixtime(todo.`modtime` ) as '指标确认时间'  FROM `kpi_process_todo` as  todo INNER JOIN `appraisal_employee_info`  as emp 
  on todo.`company_id` COLLATE utf8mb4_unicode_ci  =emp.`company_id`  and `todo`.`todo_emp_id`  = emp.`employee_id`  
  INNER JOIN  `appraisal_employee_info`  as emm on  todo.`company_id` COLLATE utf8mb4_unicode_ci  =emm.`company_id`  and `todo`.`assessee_emp_id`   = emm.`employee_id` 
  INNER JOIN `kpi_plan_info`  as plan on `todo`.`plan_id` =plan.`plan_id` 
WHERE todo.`inspection_status` =1 and todo.`is_del` =0 and todo.`company_id` ='fff4a444f84c45f588c03ad43f8883b5' and `todo`.`todo_status` =1 and plan.`plan_period`  in (7,6,30,31)  ORDER BY todo.`plan_id` ;
```
