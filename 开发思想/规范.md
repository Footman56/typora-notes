public static void main(String[] args) {
        int start = 0;
        int size = 3001;
        int end = Math.min(start + DATA_GROUP_ENTRY, size);
        do {
            System.out.println("end = " + end);
            System.out.println("start = " + start);
            start += DATA_GROUP_ENTRY;
            end = Math.min(start + DATA_GROUP_ENTRY, size);
        } while (end <= size && start < size);
    }
